==== src\App.tsx ====
import { useState, useEffect } from 'react';
import BottomNav from './components/BottomNav';
import TopBar from './components/TopBar';
import FeedView from './components/FeedView';
import ProfileView from './components/ProfileView';
import CreatePost from './components/CreatePost';
import NotificationsView from './components/NotificationsView';
import AuthView from './components/AuthView';
import { AuthProvider, useAuth } from './AuthContext';
import './index.css';

function AppShell() {
  const { currentUser } = useAuth();
  const [currentView, setCurrentView] = useState('feed');
  const [currentProfileId, setCurrentProfileId] = useState<string | null>(null);
  const [globalTheme, setGlobalTheme] = useState(localStorage.getItem('8wut-theme') || 'theme-grass');

  useEffect(() => {
    // Setup Navigation Custom Event Listener
    const handleNavigate = (e: Event) => {
      const customEvent = e as CustomEvent;
      
      if (typeof customEvent.detail === 'string') {
        setCurrentView(customEvent.detail);
        if (customEvent.detail === 'profile') setCurrentProfileId(null); // default to current user
      } else if (typeof customEvent.detail === 'object' && customEvent.detail !== null) {
        setCurrentView(customEvent.detail.view);
        if (customEvent.detail.view === 'profile' && customEvent.detail.userId) {
          setCurrentProfileId(customEvent.detail.userId);
        }
      }
      
      window.scrollTo({ top: 0, behavior: 'smooth' });
    };

    window.addEventListener('navigate', handleNavigate);

    // Setup Theme Custom Event Listener
    const handleThemeChange = (e: Event) => {
      const themeEvent = e as CustomEvent;
      setGlobalTheme(themeEvent.detail);
      localStorage.setItem('8wut-theme', themeEvent.detail);
    };

    window.addEventListener('change-theme', handleThemeChange);

    return () => {
      window.removeEventListener('navigate', handleNavigate);
      window.removeEventListener('change-theme', handleThemeChange);
    };
  }, []);

  // Apply the global theme class to document body
  useEffect(() => {
    document.body.className = '';
    document.body.classList.add(globalTheme);
  }, [globalTheme]);

  if (!currentUser) {
    return <AuthView />;
  }

  return (
    <div className={`app-shell ${globalTheme}`}>
      <TopBar currentView={currentView} />
      
      <main className={`main-content ${currentView === 'create' ? 'create-mode' : ''}`}>
        {currentView === 'feed' && <FeedView />}
        {currentView === 'eat' && <FeedView filter="fridge" />}
        {currentView === 'create' && <CreatePost />}
        {currentView === 'notifications' && <NotificationsView />}
        {currentView === 'profile' && <ProfileView userId={currentProfileId} />}
      </main>
      
      <BottomNav currentView={currentView} onViewChange={setCurrentView} />
    </div>
  );
}

export default function App() {
  return (
    <AuthProvider>
      <AppShell />
    </AuthProvider>
  );
}


==== src\AuthContext.tsx ====
import { createContext, useContext, useState, useEffect } from 'react';
import type { ReactNode } from 'react';
import type { User } from './data/mockData';
import { MOCK_USERS } from './data/mockData';

type AuthContextType = {
  currentUser: User | null;
  login: (username: string) => boolean;
  register: (username: string) => boolean;
  logout: () => void;
  updatePassword: (newPass: string) => void; 
  updateUsername: (newName: string) => void;
};

const AuthContext = createContext<AuthContextType | undefined>(undefined);

export function AuthProvider({ children }: { children: ReactNode }) {
  const [currentUser, setCurrentUser] = useState<User | null>(null);

  useEffect(() => {
    // Check localStorage on mount for a persisted user name
    const storedUser = localStorage.getItem('8wut-user');
    if (storedUser) {
      // For this demo, we'll try to match it to MOCK_USERS, or fake a user object
      const found = MOCK_USERS.find(u => u.username.toLowerCase() === storedUser.toLowerCase());
      if (found) {
        setCurrentUser(found);
      } else {
        // If they registered a new name, fake the rest of the object mock.
        setCurrentUser({
          id: 'mock-custom-' + Date.now().toString(),
          username: storedUser,
          avatarUrl: 'https://i.pravatar.cc/150?img=47',
          isFollowing: false,
          groups: []
        });
      }
    }
  }, []);

  const login = (username: string) => {
    if (!username.trim()) return false;
    
    // Check local storage mock DB for validation if needed, for KISS we just log them in
    localStorage.setItem('8wut-user', username.trim());
    
    const found = MOCK_USERS.find(u => u.username.toLowerCase() === username.trim().toLowerCase());
    if (found) {
      setCurrentUser(found);
    } else {
      setCurrentUser({
        id: 'mock-custom-' + Date.now().toString(),
        username: username.trim(),
        avatarUrl: 'https://i.pravatar.cc/150?img=47',
        isFollowing: false,
        groups: []
      });
    }
    return true;
  };

  const register = (username: string) => {
    if (!username.trim()) return false;
    // For this mock demo, registering behaves identically to login
    return login(username);
  };

  const logout = () => {
    localStorage.removeItem('8wut-user');
    setCurrentUser(null);
  };

  const updatePassword = (newPass: string) => {
    // Pure UI simulation. We don't store passwords in this mock prototype
    console.log(`Password logically updated to: ${newPass}`);
    alert("Password successfully updated. (Simulated)");
  };

  const updateUsername = (newName: string) => {
    if (!newName.trim() || !currentUser) return;
    const cleanName = newName.trim();
    setCurrentUser({
      ...currentUser,
      username: cleanName
    });
    // For pure UI simulation, we also mutate the backing store user if it was a mock user.
    const mockDbUser = MOCK_USERS.find(u => u.id === currentUser.id);
    if (mockDbUser) {
      mockDbUser.username = cleanName;
    }
    // Update local storage so a page reload remembers the new name
    localStorage.setItem('8wut-user', cleanName);
  };

  return (
    <AuthContext.Provider value={{ currentUser, login, register, logout, updatePassword, updateUsername }}>
      {children}
    </AuthContext.Provider>
  );
}

export function useAuth() {
  const context = useContext(AuthContext);
  if (context === undefined) {
    throw new Error('useAuth must be used within an AuthProvider');
  }
  return context;
}

import { MOCK_FEED } from '../data/mockData';
import PostCard from './PostCard';

export default function FeedView({ filter }: { filter?: 'fridge' }) {
  // If we are looking at the fridge, we only want fridge items.
  // The user requested the home screen to look just like the daily journal (PostCard format).
  
  let basePosts = MOCK_FEED;
  
  if (filter === 'fridge') {
    basePosts = MOCK_FEED.filter(p => p.savedToFridge);
  }

  // To maintain visual density on feed without writing much more mock data, we loop it.
  const displayPosts = [...basePosts, ...basePosts, ...basePosts].map((p, i) => ({
    ...p,
    id: `${p.id}-loop-${i}`
  }));

  return (
    <div className="feed-posts" style={{ padding: '0 8px' }}>
      {displayPosts.map(post => (
        <PostCard key={post.id} post={post} />
      ))}
      
      {filter === 'fridge' && displayPosts.length === 0 && (
         <div style={{ color: 'white', textAlign: 'center', padding: '40px 20px', opacity: 0.7 }}>
           Your fridge is empty. Save some posts!
         </div>
      )}
    </div>
  );
}

import { useState } from 'react';

export default function PostCard({ post: initialPost }: PostCardProps) {
  const [post, setPost] = useState(initialPost);
  const [activeImageIndex, setActiveImageIndex] = useState(0);
  const [showHeartAnim, setShowHeartAnim] = useState(false);
  const [isSaved, setIsSaved] = useState(post.savedToFridge || false);
  
  const [isEditing, setIsEditing] = useState(false);
  const [editCaption, setEditCaption] = useState(post.caption);
  const [isDeleted, setIsDeleted] = useState(false);

  const handleSave = () => {
    initialPost.savedToFridge = !isSaved;
    setIsSaved(!isSaved);
  };

  const handleSaveEdit = () => {
    setPost(prev => ({ ...prev, caption: editCaption }));
    setIsEditing(false);
  };

  const handleLike = () => {
    setPost(prev => ({
      ...prev,
      hasLiked: !prev.hasLiked,
      likes: prev.likes + (prev.hasLiked ? -1 : 1)
    }));
    
    if (!post.hasLiked) {
      setShowHeartAnim(true);
      setTimeout(() => setShowHeartAnim(false), 500);
